Q. 'bcaebe' 문자열을 hasOwnProperty 를 사용해서, {a : 1, b : 2, c : 1, e : 2} 로 변경해보세요.

				
var str = 'bcaebe';
var obj = {};

for (var i = 0; i < str.length; i++) {
	var cnt = 1;
	var char = str.charAt(i);
	if (obj.hasOwnProperty(char)) {
		cnt++;
		obj[char] = cnt;
	} else {
		obj[char] = cnt;
	}			
}
